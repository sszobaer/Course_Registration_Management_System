CREATE TABLE DEPARTMENT (
    DEPTID NUMBER PRIMARY KEY,
    DEPTNAME VARCHAR2(100) NOT NULL,
    DEPTHEAD VARCHAR2(100),
    OFFICELOCATION VARCHAR2(100),
    DEPTPHONE1 VARCHAR2(15),
    DEPTPHONE2 VARCHAR2(15)
);
 
CREATE TABLE DEPARTMENTHEAD (
    DEPTHEAD VARCHAR2(100) PRIMARY KEY,
    DEPTHEAD_PASSWORD VARCHAR2(100),
    DEPTID NUMBER NULL,
    FOREIGN KEY (DEPTID) REFERENCES DEPARTMENT(DEPTID)
);
 
CREATE TABLE ADMIN (
    ADMINID NUMBER PRIMARY KEY,
    ADMINNAME VARCHAR2(100) NOT NULL,
    ROLE VARCHAR2(50),
    ADMIN_PHONE1 VARCHAR2(15),
    ADMIN_PHONE2 VARCHAR2(15),
    ADMIN_EMAIL1 VARCHAR2(100),
    ADMIN_EMAIL2 VARCHAR2(100),
    DEPTID NUMBER NULL,
    FOREIGN KEY (DEPTID) REFERENCES DEPARTMENT(DEPTID)
);
 
CREATE TABLE ADMINLOGIN (
    ADMINID NUMBER PRIMARY KEY,
    ADMINNAME VARCHAR2(100),
    ADMINPASSWORD VARCHAR2(100),
    FOREIGN KEY (ADMINID) REFERENCES ADMIN(ADMINID)
);
 
CREATE TABLE STUDENT (
    STUDENTID NUMBER PRIMARY KEY,
    STUDENTNAME VARCHAR2(100) NOT NULL,
    PROGRAM VARCHAR2(50),
    ENROLLMENTDATE VARCHAR2(30),
    STATUS VARCHAR2(20),
    ADDRESS VARCHAR2(50),
    STUDENT_PHONE1 VARCHAR2(15),
    STUDENT_PHONE2 VARCHAR2(15),
    STUDENT_EMAIL1 VARCHAR2(100),
    STUDENT_EMAIL2 VARCHAR2(100),
    DEPTID NUMBER NULL,
    FOREIGN KEY (DEPTID) REFERENCES DEPARTMENT(DEPTID)
);
 
CREATE TABLE STUDENTLOGIN (
    STUDENTID NUMBER PRIMARY KEY,
    STUDENTNAME VARCHAR2(100),
    STUDENTPASSWORD VARCHAR2(100),
    FOREIGN KEY (STUDENTID) REFERENCES STUDENT(STUDENTID)
);
 
CREATE TABLE PAYMENT (
    PAYMENTID NUMBER PRIMARY KEY,
    PAYMENTDATE VARCHAR2(30) NOT NULL,
    PAYMENTMETHOD VARCHAR2(50),
    STATUS VARCHAR2(20),
    AMOUNT NUMBER(10,2),
    STUDENTID NUMBER NULL,
    FOREIGN KEY (STUDENTID) REFERENCES STUDENT(STUDENTID)
);
 
CREATE TABLE COURSE (
    COURSEID VARCHAR2(20) PRIMARY KEY,
    COURSENAME VARCHAR2(100) NOT NULL,
    CREDITS NUMBER,
    DEPTNAME VARCHAR2(50)
);
 
CREATE TABLE FACULTYCOURSEASSIGNMENT (
    FACULTYID NUMBER PRIMARY KEY,
    COURSEID VARCHAR2(20) NOT NULL,
    FOREIGN KEY (COURSEID) REFERENCES COURSE(COURSEID)
);
 
CREATE TABLE CLASSSCHEDULE (
    CLASSID NUMBER PRIMARY KEY,
    CLASSDAY VARCHAR2(20),
    CLASSTIME VARCHAR2(30),
    ROOMNO VARCHAR2(50),
    SEMESTER VARCHAR2(20),
    COURSEID VARCHAR2(20),
    CAPACITY NUMBER,
    SECTION VARCHAR2(5),
    FOREIGN KEY (COURSEID) REFERENCES COURSE(COURSEID)
);
 
CREATE TABLE FACULTYMEMBER (
    FACULTYID NUMBER PRIMARY KEY,
    FACULTYNAME VARCHAR2(100) NOT NULL,
    POSITION VARCHAR2(50),
    FACULTY_PHONE1 VARCHAR2(15),
    FACULTY_PHONE2 VARCHAR2(15),
    FACULTY_EMAIL1 VARCHAR2(100),
    FACULTY_EMAIL2 VARCHAR2(100),
    DEPTID NUMBER NULL,
    FOREIGN KEY (DEPTID) REFERENCES DEPARTMENT(DEPTID)
);
 
CREATE TABLE STUDENTCOURSEREGISTRATION (
    STUDENTID NUMBER PRIMARY KEY,
    COURSEID VARCHAR2(20) NOT NULL,
    FOREIGN KEY (STUDENTID) REFERENCES STUDENT(STUDENTID),
    FOREIGN KEY (COURSEID) REFERENCES COURSE(COURSEID)
);
 
CREATE TABLE REGISTEREDCOURSE (
    STUDENTID NUMBER,
    COURSEID VARCHAR2(20),
    SEMESTER VARCHAR2(50),
    YEAR NUMBER(4,0),
    ENROLLMENTDATE DATE,
    CLASSID NUMBER,
    FOREIGN KEY (STUDENTID) REFERENCES STUDENT(STUDENTID),
    FOREIGN KEY (COURSEID) REFERENCES COURSE(COURSEID),
    FOREIGN KEY (CLASSID) REFERENCES CLASSSCHEDULE(CLASSID)
);
 
CREATE TABLE PREREQUISITECOURSE (
    COURSEID VARCHAR2(20) PRIMARY KEY,
    PREREQUISITECOURSEID VARCHAR2(20),
    FOREIGN KEY (PREREQUISITECOURSEID) REFERENCES COURSE(COURSEID)
);
 
CREATE TABLE REGISTRATION (
    REGID NUMBER PRIMARY KEY,
    REGDATE DATE,
    STUDENTID NUMBER,
    CLASSID NUMBER,
    FOREIGN KEY (STUDENTID) REFERENCES STUDENT(STUDENTID),
    FOREIGN KEY (CLASSID) REFERENCES CLASSSCHEDULE(CLASSID)
);
